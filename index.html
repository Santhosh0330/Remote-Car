Introduction

This project is about building a multi-function car using an Arduino Uno, an L298N motor driver, four DC motors, and a 9V battery for power. The parts are wired together on a breadboard so the Arduino can control the motors and make the car move. The goal is to assemble the circuit and test that the car works properly.

Circuit Diagram

This circuit shows an Arduino Uno connected to two L298N motor driver on a breadboard, powered by a 9V battery. The motor driver is wired to control four DC motors, including two main drive motors and two additional motors for extra movement or functions. Power and control lines run from the Arduino to the motor driver, allowing the system to manage motor speed and direction. The layout distributes power across the breadboard while supplying enough current for all four motors to operate together.

Arduino Code

The following Arduino code controls the system, lighting LEDs and activating the buzzer based on distance readings from the HC-SR04 sensor.

  
  

// Multi Function Car
// Author: Massimo Solen Adam Santhosh
// November 26th 2025
// Remote controlled car with different functions like headlights and break lights.

// --------------------
// Pin Assignments
// --------------------

// LEFT SIDE (L298N #1)
#define L1_IN1 5
#define L1_IN2 4
#define L2_IN1 6
#define L2_IN2 7

// RIGHT SIDE (L298N #2)
#define R1_IN1 9
#define R1_IN2 8
#define R2_IN1 10
#define R2_IN2 11

// JOYSTICKS
#define JOY_LEFT A0
#define JOY_RIGHT A1

// Dead zone to prevent motor jitter
int deadzone = 50;

// --------------------
// Helper: Set motor speed and direction
// --------------------
void driveMotor(int in1, int in2, int speed) {
  speed = constrain(speed, -255, 255);

  if (speed > 0) {
    analogWrite(in1, speed);
    analogWrite(in2, 0);
  } else if (speed < 0) {
    analogWrite(in1, 0);
    analogWrite(in2, -speed);
  } else {
    analogWrite(in1, 0);
    analogWrite(in2, 0);
  }
}

void setup() {
  pinMode(L1_IN1, OUTPUT);
  pinMode(L1_IN2, OUTPUT);
  pinMode(L2_IN1, OUTPUT);
  pinMode(L2_IN2, OUTPUT);

  pinMode(R1_IN1, OUTPUT);
  pinMode(R1_IN2, OUTPUT);
  pinMode(R2_IN1, OUTPUT);
  pinMode(R2_IN2, OUTPUT);

  Serial.begin(9600);
}

void loop() {
  // --------------------
  // Read Joysticks
  // --------------------
  int joyL = analogRead(JOY_LEFT);
  int joyR = analogRead(JOY_RIGHT);

  // Convert 0–1023 → -255 to +255
  int leftPower  = map(joyL, 0, 1023, 255, -255);
  int rightPower = map(joyR, 0, 1023, 255, -255);

  // Deadzone
  if (abs(leftPower) < deadzone) leftPower = 0;
  if (abs(rightPower) < deadzone) rightPower = 0;

  // --------------------
  // LEFT TRACK MOTORS
  // --------------------
  driveMotor(L1_IN1, L1_IN2, leftPower);
  driveMotor(L2_IN1, L2_IN2, leftPower);

  // --------------------
  // RIGHT TRACK MOTORS
  // --------------------
  driveMotor(R1_IN1, R1_IN2, rightPower);
  driveMotor(R2_IN1, R2_IN2, rightPower);

  // Debug (optional)
  Serial.print("L: "); Serial.print(leftPower);
  Serial.print("  R: "); Serial.println(rightPower);

  delay(10);
}
